create table en_clinica (
	cod_clinica serial not null,
	dsc_endereco_clinica char(50) not null,
	primary key (cod_clinica)
);

create table en_local (
	cod_local serial not null,
	desc_endereco_local char(50) not null,
	primary key (cod_local)
);

create table en_remedios (
	cod_remedio serial not null,
	dsc_remedio char(100),
	nme_remedio char(100) not null,
	primary key (cod_remedio)
);

create table en_especialidade (
	cod_especialidade serial not null,
	nme_especialidade char(100) not null,
	dsc_especialidade char(100),
	primary key (cod_especialidade)
);

create table en_convenio_utilizacao (
	cod_clinica serial not null,
	cod_especialidade serial not null,
	cod_local serial not null,
	foreign key (cod_clinica) references en_clinica,
	foreign key (cod_especialidade) references en_especialidade,
	foreign key (cod_local) references en_local,
	primary key (cod_clinica, cod_especialidade, cod_local)
);

create table en_medico (
	cod_medico serial not null,
	cod_clinica serial not null,
	cod_especialidade serial not null,
	nme_medico char(50) not null,
	dsc_endereco_medico char(50) not null,
	salario_medico float not null,
	foreign key (cod_clinica) references en_clinica,
	foreign key (cod_especialidade) references en_especialidade,
	primary key (cod_medico, cod_clinica)
);

create table en_paciente (
	cod_paciente serial not null,
	nme_paciente char(50) not null,
	sexo_paciente char(15) not null,
	dta_nascimento_paciente date not null,
	dsc_endereco_paciente char(50),
	primary key (cod_paciente)
);

create table en_receita (
	cod_medico serial not null,
	cod_medico_clinica serial not null,
	cod_clinica serial not null,
	cod_especialidade serial not null,
	cod_local serial not null,
	cod_paciente serial not null,
	cod_remedio serial not null,
	foreign key (cod_medico, cod_medico_clinica) references en_medico,
	foreign key (cod_clinica) references en_clinica,
	foreign key (cod_especialidade) references en_especialidade,
	foreign key (cod_local) references en_local,
	foreign key (cod_paciente) references en_paciente,
	foreign key (cod_remedio) references en_remedios,
	primary key (cod_medico, cod_medico_clinica, cod_clinica, cod_especialidade, 
	cod_local, cod_paciente, cod_remedio)
);

create table en_consulta (
	cod_medico serial not null,
	cod_medico_clinica serial not null,
	cod_clinica serial not null,
	cod_especialidade serial not null,
	cod_local serial not null,
	cod_paciente serial not null,
	dta_consulta date not null,
	valor_consulta float not null,
	foreign key (cod_medico, cod_medico_clinica) references en_medico,
	foreign key (cod_clinica) references en_clinica,
	foreign key (cod_especialidade) references en_especialidade,
	foreign key (cod_local) references en_local,
	foreign key (cod_paciente) references en_paciente,
	primary key (cod_medico, cod_medico_clinica, cod_clinica, cod_especialidade, cod_local, cod_paciente)
);
